⚙️ Live Deployment Guide

# Dreame GPT 프로덕션 초기화
export DREAME_ENV=production
export DREAME_API_KEY=$(cat ~/.dreame/token.key)
source ~/.bash_profile

mkdir -p /opt/dreame-gpt/{config,data,logs,tmp,cache}
cp ./config/system_config.yaml /opt/dreame-gpt/config/
cp ./data/faq_dataset.xlsx /opt/dreame-gpt/data/

python3 scripts/init_vector_db.py --config /opt/dreame-gpt/config/system_config.yaml --force-reindex

systemctl restart dreame-gpt.service
tail -n 100 -f /var/log/dreame-gpt/inference.log | grep -iE "inference|latency"

curl -s -X GET http://localhost:8080/healthcheck -H "X-Auth-Token: $DREAME_API_KEY" | jq .status
